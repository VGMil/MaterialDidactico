# Arquero

La declaración <Highlight color="green">public class Arquero extends PersonajeModel</Highlight> <br/> indica que Arquero hereda todos los <Highlight color="green">atributos y métodos</Highlight> de la clase abstracta <Highlight color="green">PersonajeModel</Highlight>. Esto garantiza que todo Arquero tenga las propiedades esenciales de un <Highlight color="green">personaje</Highlight> (Vida, Ataque Físico, Defensa, etc.).

{`
import Personaje.Models.PersonajeModel;

public class Arquero extends PersonajeModel { 
    
    private static final int ArqueroVida = 1500; 
    private static final int ArqueroAtaqueFisico = 80;
    private static final int ArqueroDefensa = 40;
    private static final int ArqueroVelocidad = 90;

    public Arquero(String nombre) {
    // Llama al constructor de la superclase (PersonajeModel)
        super(ArqueroVida, ArqueroAtaqueFisico, ArqueroDefensa, ArqueroVelocidad, nombre);
    }   

    @Override
    public void interactuar() {
        System.out.println("El Arquero " + this.getNombre() + " observa desde la distancia.");
    }
    @Override
    public void atacar(PersonajeModel enemigo) {
        // 1. Imprimimos el mensaje de acción
        System.out.println("El Arquero " + this.getNombre() + " dispara una flecha certera a "
        + enemigo.getNombre() + "!");

        // 2. Calculamos el daño efectivo (Daño = Ataque Físico - Defensa del Enemigo)
        int dañoInfligido = this.ataqueFisico - enemigo.getDefensa();
    
        // Aseguramos que el daño sea al menos 1, no negativo
        if (dañoInfligido < 1) {
            dañoInfligido = 1; 
        }

        // 3. Aplicamos el daño (Restamos el daño a la vida del enemigo)
        enemigo.setVida(enemigo.getVida() - dañoInfligido);

        // 4. Imprimimos el resultado de la acción
        System.out.println("¡Ha infligido " + dañoInfligido +
         " puntos de daño!");
        System.out.println("La vida restante de " + enemigo.getNombre() +
         " es: " + enemigo.getVida());
    }

    @Override
    public void escapar(PersonajeModel enemigo) {
        System.out.println("El Arquero " + this.getNombre() + " se desvanece entre las sombras dejando atrás a " + enemigo.getNombre());
    }

}
