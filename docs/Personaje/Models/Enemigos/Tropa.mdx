# Tropa

La declaración <Highlight color="green">public class Tropa extends PersonajeModel</Highlight> <br/> indica que Tropa hereda todos los <Highlight color="green">atributos y métodos</Highlight> de la clase abstracta <Highlight color="green">PersonajeModel</Highlight>. Esto garantiza que toda Tropa tenga las propiedades esenciales de un <Highlight color="green">personaje</Highlight> (Vida, Ataque, Defensa, etc.).

import Personaje.Models.PersonajeModel;

public class Tropa extends PersonajeModel { 
    
    private static final int TropaVida = 500; 
    private static final int TropaAtaque = 30;
    private static final int TropaDefensa = 20;
    private static final int TropaVelocidad = 50;

    public Tropa(String nombre) {
    // Llama al constructor de la superclase (PersonajeModel)
        super(TropaVida, TropaAtaque, TropaDefensa, TropaVelocidad, nombre);
    }   

    @Override
    public void interactuar() {
        System.out.println(this.getNombre() + " patrulla la zona con mirada vigilante.");
    }
    @Override
    public void atacar(PersonajeModel enemigo) {
        // 1. Imprimimos el mensaje de acción
        System.out.println(this.getNombre() + " ataca rápidamente a "
        + enemigo.getNombre() + ".");

        // 2. Calculamos el daño efectivo
        int dañoInfligido = this.ataque - enemigo.getDefensa();
    
        // Aseguramos que el daño sea al menos 1, no negativo
        if (dañoInfligido < 1) {
            dañoInfligido = 1; 
        }

        // 3. Aplicamos el daño
        enemigo.setVida(enemigo.getVida() - dañoInfligido);

        // 4. Imprimimos el resultado de la acción
        System.out.println("¡Ha infligido " + dañoInfligido +
         " puntos de daño!");
        System.out.println("La vida restante de " + enemigo.getNombre() +
         " es: " + enemigo.getVida());
    }

    @Override
    public void escapar(PersonajeModel enemigo) {
        System.out.println(this.getNombre() + " rompe filas y huye despavorido ante " + enemigo.getNombre());
    }

}
