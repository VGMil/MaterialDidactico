# Curandero

La declaración <Highlight color="green">public class Curandero extends PersonajeModel</Highlight> <br/> indica que Curandero hereda todos los <Highlight color="green">atributos y métodos</Highlight> de la clase abstracta <Highlight color="green">PersonajeModel</Highlight>. Esto garantiza que todo Curandero tenga las propiedades esenciales de un <Highlight color="green">personaje</Highlight> (Vida, Ataque Físico, Defensa, etc.).

import Personaje.Models.PersonajeModel;

public class Curandero extends PersonajeModel { 
    
    private static final int CuranderoVida = 1800; 
    private static final int CuranderoAtaqueFisico = 10;
    private static final int CuranderoDefensa = 50;
    private static final int CuranderoVelocidad = 60;

    public Curandero(String nombre) {
    // Llama al constructor de la superclase (PersonajeModel)
        super(CuranderoVida, CuranderoAtaqueFisico, CuranderoDefensa, CuranderoVelocidad, nombre);
    }   

    @Override
    public void interactuar() {
        System.out.println("El Curandero " + this.getNombre() + " prepara sus vendas y pociones.");
    }
    @Override
    public void atacar(PersonajeModel enemigo) {
        // 1. Imprimimos el mensaje de acción
        System.out.println("El Curandero " + this.getNombre() + " golpea con su bastón a "
        + enemigo.getNombre() + "!");

        // 2. Calculamos el daño efectivo (Daño = Ataque Físico - Defensa del Enemigo)
        int dañoInfligido = this.ataqueFisico - enemigo.getDefensa();
    
        // Aseguramos que el daño sea al menos 1, no negativo
        if (dañoInfligido < 1) {
            dañoInfligido = 1; 
        }

        // 3. Aplicamos el daño (Restamos el daño a la vida del enemigo)
        enemigo.setVida(enemigo.getVida() - dañoInfligido);

        // 4. Imprimimos el resultado de la acción
        System.out.println("¡Ha infligido " + dañoInfligido +
         " puntos de daño!");
        System.out.println("La vida restante de " + enemigo.getNombre() +
         " es: " + enemigo.getVida());
    }

    @Override
    public void escapar(PersonajeModel enemigo) {
        System.out.println("El Curandero " + this.getNombre() + " pide clemencia y se retira ante " + enemigo.getNombre());
    }

}
